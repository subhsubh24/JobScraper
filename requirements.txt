# Career Operator — LEAN runtime deps for the deployed serverless API (Vercel).
# Keep this small: Vercel's Python function bundle has a ~250MB unzipped limit.
# Anything not imported by asgi.py + src/* belongs in requirements-dev.txt, NOT here.
# (sentence-transformers/torch, celery, redis, pandas, etc. were removed — unused by
#  the API and they blow the size limit. See docs/DEPLOY_VERCEL.md.)

# Web API framework
fastapi==0.109.0
uvicorn[standard]==0.27.0          # local dev / container fallback; Vercel wraps ASGI itself

# Database (Postgres in production — SQLite has no persistence on serverless)
sqlalchemy==2.0.25
psycopg2-binary==2.9.9

# LLM (optional at runtime — degrades gracefully without OPENAI_API_KEY)
openai==1.59.6  # >=1.55 required: 1.10 breaks with httpx>=0.28 ("proxies" TypeError) on Vercel

# Billing — Stripe Checkout + webhook signature verification (Track C). Pure-Python,

# src/billing.py + the /api/billing/* endpoints, so it must ship in the runtime bundle.
stripe==7.9.0

# HTTP client — PINNED to keep prod==dev (transitive via openai; starlette TestClient too).
# httpx 0.28 dropped the Client(app=...) kwarg that starlette 0.35's TestClient still
# passes, so an unpinned httpx (floats to 0.28+) silently breaks the entire test/journey
# suite while prod looks fine. Pin <0.28 until fastapi/starlette are bumped to a release
# whose TestClient uses ASGITransport. openai 1.59.6 supports httpx 0.27 (>=0.23,<1).
httpx==0.27.2

# Core utilities actually imported by the API
